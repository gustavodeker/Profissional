CREATE TABLE users (
    user_id int PRIMARY KEY AUTO_INCREMENT,
    user_name varchar(50) UNIQUE NOT NULL,
    user_pass varchar(255) NOT NULL,
    user_level varchar(50) NOT NULL,
    user_token varchar(255) NULL
);

CREATE TABLE codes (
	code_id int PRIMARY KEY AUTO_INCREMENT,
    code_code varchar(50) UNIQUE NOT NULL,
    code_desc varchar(255) NOT NULL
);

CREATE TABLE records (
	record_id int PRIMARY KEY AUTO_INCREMENT,
    record_user_id int(11) NOT NULL,
    record_code_id int(11) NOT NULL,
    record_value int(11) NOT NULL,
    record_time varchar(255) NOT NULL,
    FOREIGN KEY (record_user_id) REFERENCES users(user_id),
    FOREIGN KEY (record_code_id) REFERENCES codes(code_id)
);
